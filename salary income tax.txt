    private void calcActionPerformed(java.awt.event.ActionEvent evt) {                                     
        // TODO add your handling code here:
        double salary=Double.parseDouble(txtsalary.getText());
        double tax, net, ep;
                if(salary >=10900){
     tax=salary*35/100-1500;
}
else if(salary>=7801){
    tax=salary*30/100-955;
  }
else if(salary>=5251){
  tax=salary*25/100-565;
}
else if(salary>=3201){
  tax=salary*20/100-302.5;
}

else if(salary>=1651){
  tax=salary*15/100-142.50;
}
else if(salary>=601){
  tax=salary*10/100-60;
}
else{
  tax=0.00;
}
                txttax.setText(String.valueOf(tax));
                ep=salary*7/100;
                txtep.setText(String.valueOf(ep));
                net = salary-tax-ep;
                txtneti.setText(String.valueOf(net));
    }
