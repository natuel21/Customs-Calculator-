         gst = price *15/100;
         txtat.setText(String.valueOf(gst));
         nat= price + gst;
         txtna.setText(String.valueOf(nat));
         gst1 =  price - (price * 0.87);
           txtrt.setText(String.valueOf(gst1));
           natr = price -gst1;
             txtnr.setText(String.valueOf(natr));
