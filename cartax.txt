    private void calcActionPerformed(java.awt.event.ActionEvent evt) {                                     

        double price =Double.parseDouble(pricec.getText());
         double cts,ets, vats ,stss,tts,ngs;
         cts= price*30/100;
         ct.setText(String.valueOf(cts));
         ets= (price + cts )*5/100;
         et.setText(String.valueOf(ets));
         vats= (price + cts + ets) *15/100;
         vat.setText(String.valueOf(vats));
         stss = (price + ets + cts + vats )*10/100;
         st.setText(String.valueOf(stss));
         tts= price+cts+ets+vats+stss;
         tp.setText(String.valueOf(tts));
         ngs= cts+ets+vats+stss;
         ngp.setText(String.valueOf(ngs));
    }
